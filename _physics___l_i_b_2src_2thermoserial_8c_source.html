51double _get_value_1D(double time_step,
52 double space_step,
53 double x, double t,
54 int precision)
55{
56 double sum = 0.0, exponential, spaceXTerm, coeff;
57 double x_real = x * space_step;
58 double t_real = t * time_step;
59
60 for (int k = 0; k < precision; k++)
61 {
62 exponential = exp(-3 * pow(2 * (k + 1), 2) * (M_PI * M_PI * t_real) / 4);
63 spaceXTerm = sin((double)(2 * k + 1) * M_PI * x_real / 2);
64 coeff = 1 / (2 * k + 1);
65 sum += coeff * exponential * spaceXTerm;
66 }
67
68 sum *= 200 / M_PI;
69 return sum;
70}

89double _get_value_2D(double time_step,
90 double length, double space_step_x, double width, double space_step_y,
91 int x, int y, int t,
92 int precision) {
93 double sum = 0.0, exponential, spaceXTerm, spaceYTerm, coeff;
94 double x_real = x * space_step_x;
95 double y_real = y * space_step_y;
96 double t_real = t * time_step;
97 for (ll m = 1; m < precision; ++m) {
98 for (ll n = 1; n < precision; ++n) {
99 exponential = exp(-(M_PI * M_PI) * (m * m + n * n) * t_real / 36);
100 spaceXTerm = sin((double) m * M_PI * x_real / length);
101 spaceYTerm = sin((double) n * M_PI * y_real / width);
102 // Find Amn constant and multiply it with the sum
103 coeff = (1 + pow(-1, m + 1)) * (1 - cos(n * M_PI / 2)) / (m * n);
104 sum += coeff * exponential * spaceXTerm * spaceYTerm;
105 }
106 }
107 sum *= 200 / (M_PI * M_PI);
108 return sum;
109}
