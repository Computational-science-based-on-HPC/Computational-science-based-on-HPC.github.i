81_simulate_elastic_pendulum(double r, double length, double mass, double gravity, double k, double Ao, double Xo,
82 double Yo,
83 double Vo,

90 puts("Invalid Arguments is Given");
91 return -1;
92 }

99 double x1=Xo; // init position of mass in x
100 double y=Yo; // init position of mass in y

102 double a=0; // init velocity
103 for(int i=0; i<time_limit; ++i)
104 {
105 t+=step_size;
106 double k11=_dx(v);
107 double k12=_dy(a);
108 double k13=_f1(x1, y, v, 0, 0, k, mass, damping_coefficent, r);

112 double k22=_dy(a + step_size/2*k14);
113 double k23=_f1(x1 + step_size/2*k11, y + step_size/2*k12, v + step_size/2*k13, 0, 0, k, mass,
114 damping_coefficent, r);
115 double k24=_f2(x1 + step_size/2*k11, y + step_size/2*k12, a + step_size/2*k14, 0, 0, k, mass,
116 damping_coefficent, r, gravity);
117
118 double k31=_dx(v + step_size/2*k23);
119 double k32=_dy(a + step_size/2*k24);
120 double k33=_f1(x1 + step_size/2*k21, y + step_size/2*k22, v + step_size/2*k23, 0, 0, k, mass,
121 damping_coefficent, r);
122 double k34=_f2(x1 + step_size/2*k21, y + step_size/2*k22, a + step_size/2*k24, 0, 0, k, mass,
123 damping_coefficent, r, gravity);
124
125 double k41=_dy(v + step_size/2*k33);
126 double k42=_dy(a + step_size/2*k34);
127 double k43=_f1(x1 + step_size*k31, y + step_size*k32, v + step_size*k33, 0, 0, k, mass,

130 damping_coefficent, r, gravity);
131 x1+=step_size/6.0*(k11 + 2*k21 + 2*k31 + k41);
132 y+=step_size/6.0*(k12 + 2*k22 + 2*k32 + k42);
133 v+=step_size/6.0*(k13 + 2*k23 + 2*k33 + k43);
134 a+=step_size/6.0*(k14 + 2*k24 + 2*k34 + k44);
135 }
